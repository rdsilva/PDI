Atividade 3.2 A

  • Nesta atividade deve-se solicitar ao usuário uma imagem e carregá-la, em seguida deve-se perguntar para o usuário por duas coordenadas de uma região dentro da imagem e negativá-la. Para a minha solução o usuário deverá informar alguma imagem e então utilizar o próprio mouse para gerar a região desejada, bastando clicar e arrastar entre o ponto inicial desejado e o ponto final.

///////////////////////////////////////////////////////////////////////////////////////////////////
/*
Este metodo reconhece as coordenadas informadas pelo click do mouse
Verifica quais s„o os pontos maximo e minimos para X e Y
Percorre a matriz partindo do ponto minimo ao maximo invertendo o valor do pixel

A invers„o do valor do pixel d·-se pelo complemento de 255, ou seja
valor_novo = 255 - valor_atual
*/
void Negativar() {

	std::cout << "\n-------- Negativando a ROI definida...\n";

	int max_X = int(fmax(px_1, px_2));
	int max_Y = int(fmax(py_1, py_2));
	int min_X = int(fmin(px_1, px_2));
	int min_Y = int(fmin(py_1, py_2));

	// criando a regi„o com negativo da imagem
	cv::cvtColor(imgOriginal, imgResult, CV_BGR2GRAY);

	for (int i = min_X; i<max_X; i++) {
		for (int j = min_Y; j<max_Y; j++) {
			imgResult.at<uchar>(j, i) = 255 - imgResult.at<uchar>(j, i);
		}
	}

	cv::imshow("imgResult", imgResult);

	std::cout << "-------- ROI Negativada...\n\n";

}

///////////////////////////////////////////////////////////////////////////////////////////////////
/*
O mÈtodo de CallBack trata o uso de eventos do mouse sobre uma imagem
Neste caso ele È respons·vel armazenar os pontos de clique do mouse e
por chamar o metodo Negativar, respons·vel por gerar o negativo da ROI
*/

void CallBackFunc(int event, int x, int y, int flags, void* userdata){
	// Primeiro ponto, quando o bot„o esquerdo do mouse È clicado
	if (event == cv::EVENT_LBUTTONDOWN && !clicado){
		px_1 = x;
		py_1 = y;
		clicado = true;
		std::cout << "Mouse clicou na posicao 1 - (" << x << ", " << y << ")" << std::endl;
	}
	// Segundo ponto, quando o bot„o esquerdo do mouse È liberado
	if (event == cv::EVENT_LBUTTONUP && clicado){
		px_2 = x;
		py_2 = y;
		clicado = false;
		std::cout << "Mouse arrastado ate a posicao 2 - (" << x << ", " << y << ")" << std::endl;

		Negativar();
	}
}

Atividade 3.2 B

  • Nesta atividade deve-se usar o mesmo principio de seleção de uma região da imagem, contudo, deve-se carregar uma imagem trocar as regiões desta de forma aleatória. A minha solução calcula 4 ROIs (Regions of Interest - Regiões de Interesse) da imagem, de forma automatica, e de forma randomizada calcula a ordem em que estas serão dispostas, formando uma nova imagem e seguindo a distribuição das áreas em Z.

Atividade 5.2 A

  • Esta atividade já é um pouco mais evoluída que as demais, forçando o aluno a trabalhar agora com vídeo. Apesar de um vídeo não ser nada mais que uma sequência de imagens, o processamento em tempo real é extremamente sensível. Para esta atividade o aluno deverá equalizar em tempo real a imagem que é capturada. Para a minha solução eu capturo o quadro do vídeo (uma imagem), converto para YCbCr, aplico a equalização apenas no canal Y e reconverto a imagem para RGB. A grande sacada é que equalizar uma imagem nada mais é que equalizar os níveis de intensidade luminosa desta, ou seja, como o canal Y é o canal da intensidade luminosa, equalizando este eu equalizo a imagem.

#include <iostream>
#include <opencv2\opencv.hpp>
#include <opencv2\highgui\highgui.hpp>
#include <opencv2\imgproc\imgproc.hpp>

using namespace cv;
using namespace std;

int width, height;
int nbins = 64;
int histw = nbins, histh = nbins / 2;
float range[] = { 0, 256 };
const float *histrange = { range };
bool uniform = true;
bool acummulate = false;

Mat histograma(Mat image, vector<Mat> planes, Mat histImgR, Mat histImgG, Mat histImgB, Mat histR, Mat histG, Mat histB) {
	// calculando o histograma da imagem original
	calcHist(&planes[0], 1, 0, Mat(), histR, 1,
		&nbins, &histrange,
		uniform, acummulate);
	calcHist(&planes[1], 1, 0, Mat(), histG, 1,
		&nbins, &histrange,
		uniform, acummulate);
	calcHist(&planes[2], 1, 0, Mat(), histB, 1,
		&nbins, &histrange,
		uniform, acummulate);

	// normalizando o histograma
	normalize(histR, histR, 0, histImgR.rows, NORM_MINMAX, -1, Mat());
	normalize(histG, histG, 0, histImgG.rows, NORM_MINMAX, -1, Mat());
	normalize(histB, histB, 0, histImgB.rows, NORM_MINMAX, -1, Mat());

	histImgR.setTo(Scalar(0));
	histImgG.setTo(Scalar(0));
	histImgB.setTo(Scalar(0));

	// linearizando o historgama
	for (int i = 0; i<nbins; i++) {
		line(histImgR, Point(i, histh),
			Point(i, cvRound(histR.at<float>(i))),
			Scalar(0, 0, 255), 1, 8, 0);
		line(histImgG, Point(i, histh),
			Point(i, cvRound(histG.at<float>(i))),
			Scalar(0, 255, 0), 1, 8, 0);
		line(histImgB, Point(i, histh),
			Point(i, cvRound(histB.at<float>(i))),
			Scalar(255, 0, 0), 1, 8, 0);
	}

	// apresentando o histograma
	histImgR.copyTo(image(Rect(0, 0, nbins, histh)));
	histImgG.copyTo(image(Rect(0, histh, nbins, histh)));
	histImgB.copyTo(image(Rect(0, 2 * histh, nbins, histh)));

	return image;
}

Atividade 5.2 B

  • Esta atividade é um desafio um pouco maior que a atividade anterior. Deve-se monitorar uma imagem e detectar movimento nesta quando ocorrer. Utilizando basicamente o mesmo principio da atividade anterior, converto a imagem para o modelo HSV, crio uma ROI que será utilizada para monitorar os canais H (hue - valor) e S (saturation - saturação), calculo o histograma para a imagem capturada e para a ROI, utilizo o algoritmo nativo do OpenCV Bhattacharyya para calcular a correlação entre os histogramas, e calculei empiricamente um threshold para detectar um movimento suave. Para o threshold quanto menor o valor resultado mais sensível ao movimento será este algoritmo.

#include <iostream>
#include <opencv2/opencv.hpp>

using namespace cv;
using namespace std;

int main(int argc, char** argv) {
	Mat image, imageHSV;
	int width, height;
	int count = 0;
	double last_value = 0.0;
	double threshold = 1.6;
	VideoCapture cap;

	cap.open(0);

	if (!cap.isOpened()) {
		cout << "cameras indisponiveis";
		return -1;
	}

	width = cap.get(CV_CAP_PROP_FRAME_WIDTH);
	height = cap.get(CV_CAP_PROP_FRAME_HEIGHT);

	cout << "largura = " << width << endl;
	cout << "altura  = " << height << endl;

	while (1) {
		cap >> image;

		cvtColor(image, imageHSV, COLOR_BGR2HSV);

                // criando uma ROI para comparação
		Mat hsv_half_down = imageHSV(Range(imageHSV.rows / 2, imageHSV.rows - 1), Range(0, imageHSV.cols - 1));

		// usando 50 pontos para o canal H e 60 para o S
		int h_bins = 50; int s_bins = 60;
		int histSize[] = { h_bins, s_bins };

		// range de variação para H e S
		float h_ranges[] = { 0, 180 };
		float s_ranges[] = { 0, 256 };

		const float* ranges[] = { h_ranges, s_ranges };

		// usando apenas os canais H e S
		int channels[] = { 0, 1 };

		// histogramas
		MatND hist_base;
		MatND hist_half_down;

		// calculando os histogramas da imagem original e da ROI
		calcHist(&imageHSV, 1, channels, Mat(), hist_base, 2, histSize, ranges, true, false);
		normalize(hist_base, hist_base, 0, 1, NORM_MINMAX, -1, Mat());

		calcHist(&hsv_half_down, 1, channels, Mat(), hist_half_down, 2, histSize, ranges, true, false);
		normalize(hist_half_down, hist_half_down, 0, 1, NORM_MINMAX, -1, Mat());

		// fazendo a comparação enter os histogramas

		int compare_method = 4;
		double base_half = compareHist(hist_base, hist_half_down, compare_method);

		double test = abs(base_half - last_value);

		if (test > threshold) {
			printf("%3d --> MOVIMENTO \n", count);
			last_value = base_half;
			count++;
		}

		imshow("image", image);
		if (waitKey(30) >= 0) break;
	}
	return 0;
}

Atividade 8.1

  • Esta atividade tinha por objetivo aplicar conceitos que não são necessariamente aprendidas apenas com a disciplinad e Processamento Digital de Imagens. A Transformada de Fourier de um sinal, seja ele de qualquer natureza, permite um processamento menos "burocrático", uma vez que podemos realizar procedimentos matemáticos mais simples.

  • O objetivo aqui é corrigir possíveis falhas de iluminação na imagem, assim é necessário que sejam trabalhadas as propriedades de reflectância e iluminação da imagem, respectivamente associados às frequências altas e baixas. Para realizar esta atividade é necessário primeiramente aplicar a função Log na imagem, assim conseguimos obter as duas caracteristicas citadas anteriormente nas quais precisaremos trabalhar. Depois aplicamos a transformada de Fourier e então um filtro que amplifique as altas frequências e atenue as baixas frequências. Depois de todo o processo aplicamos a inversa da transformada de Fourier e um merge das camadas para que obtenhamos a imagem resultado.

///////////////////////////////////////////////////////////////////////////////////////////////////
/*
Filtro passa alta, implementado conforme leitura do livro do Gonzalez
*/
void filtro(cv::Mat& input, cv::Mat& output, float lower, float upper, float threshold) {
	cv::Mat* i_ptr = &input;
	int width = i_ptr->cols;
	int height = i_ptr->rows;
	int channel = i_ptr->channels();

	cv::Mat* o_ptr = &output;
	if (i_ptr != o_ptr) {
		*o_ptr = cv::Mat(height, width, CV_MAKETYPE(CV_32F, channel));
	}

	for (int y = 0; y<height; y++) {
		for (int x = 0; x<width; x++) {
			float r = sqrt((float)(x*x + y*y));
			double coeff = (1.0 - 1.0 / (1.0 + exp(r - threshold))) * (upper - lower) + lower;
			for (int c = 0; c<channel; c++) {
				o_ptr->at<float>(y, x*channel + c) = coeff * i_ptr->at<float>(y, x*channel + c);
			}
		}
	}
}

		// Aplicando a DFT, o filtro e a inversa da DFT
		vector<cv::Mat> chs, spc(channel, cv::Mat(height, width, CV_32FC1));
		cv::split(imgOriginal, chs);

		for (int c = 0; c<channel; c++) {
			cv::dct(chs[c], spc[c]);
			filtro(spc[c], spc[c], lower, upper, threshold);
			cv::idct(spc[c], chs[c]);
		}
		cv::merge(chs, imgResult);

		// Plotando os parametros na tela

		system("cls");
		std::cout << "\nUse A para aumentar o Gama Low ou Z para diminuir";
		std::cout << "\nUse S para aumentar o Gama High ou X para diminuir";
		std::cout << "\nUse D para aumentar o Threshold ou C para diminuir";
		std::cout << "\n";
		std::cout << "\nGama Low \t: " << lower;
		std::cout << "\nGama High \t: " << upper;
		std::cout << "\nThreshold \t: " << threshold;


		// Apresentando os resultados
		cv::imshow("Original", imgOriginal);
		cv::imshow("Resultado", imgResult);

		char key = (char) cv::waitKey(10);
		if (key == 27) loop = false; // esc pressed!

		switch (key) {
			case 'a': //aumenta a escala para baixas frequencias
				lower += 0.05f;
				break;
			case 'z': //diminui a escala para baixas frequencias
				lower -= 0.05f;
				break;
			case 's': //aumenta a escala para altas frequencias
				upper += 0.05f;
				break;
			case 'x': //diminui a escala para altas frequencias
				upper -= 0.05f;
				break;
			case 'd': //aumenta o valor de threshold
				threshold += 0.05f;
				break;
			case 'c': //diminui o valor de threshold
				threshold -= 0.05f;
				break;
			default:
				break;
		}

	}

	cv::destroyAllWindows();
	return(0);

}

/*
Algoritmo de Pontilhismo
*/
#define STEP 2
#define JITTER 1
int RAIO = 2;

void pontilhismo(Mat border) {
	vector<int> yrange;
	vector<int> xrange;

	Mat imagePontilhismo, imageBase;

	int width, height, gray;
	int x, y;

	border.copyTo(imagePontilhismo);
	//cvtColor(border, imagePontilhismo, CV_BGR2GRAY);
	cvtColor(imgOriginal, imageBase, CV_BGR2GRAY);

	srand(time(0));

	width = imagePontilhismo.size().width;
	height = imagePontilhismo.size().height;

	xrange.resize(height / STEP);
	yrange.resize(width / STEP);

	iota(xrange.begin(), xrange.end(), 0);
	iota(yrange.begin(), yrange.end(), 0);

	for (uint i = 0; i<xrange.size(); i++) {
		xrange[i] = xrange[i] * STEP + STEP / 2;
		yrange[i] = yrange[i] * STEP + STEP / 2;
	}

	imgResult = Mat(height, width, CV_8U, Scalar(255));

	random_shuffle(xrange.begin(), xrange.end());

	for (auto i : xrange) {
		random_shuffle(yrange.begin(), yrange.end());
		for (auto j : yrange) {
			x = i + rand() % (2 * JITTER) - JITTER + 1;
			y = j + rand() % (2 * JITTER) - JITTER + 1;
			gray = imageBase.at<uchar>(x, y); // <-- aqui tenho que inserir o borderCanny para referencia de onde pintar as bolinhas (quanto maior o threshold do Canny ... maior o diametro das bolinhas)

			if (imagePontilhismo.at<uchar>(x, y) == 0) {
				RAIO = 1;
			} else {
				RAIO = 2;
			}

			circle(imgResult,
				cv::Point(y, x),
				RAIO,
				CV_RGB(gray, gray, gray),
				-1,
				CV_AA);
		}
	}


	namedWindow("Resultado", CV_WINDOW_AUTOSIZE);
	imshow("Resultado", imgResult);
}
